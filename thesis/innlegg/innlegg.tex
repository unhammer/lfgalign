      \begin{itemize}

       \item berre lenkjer som samsvarer med ein dependensanalyse \citep{hearne2008ccd}

      \end{itemize}
    \end{itemize}
  \end{frame}

    \item Formål: annotert trebank
    \item Kunnskapsdriven lenking, kan ev. filtrerast med N-gramtabell
      (eller omsetjingsordbok)
    \end{itemize}
  \end{itemize}
